# Show the specgram of the wav file.
y=zeros(numframes)
for i in range(numframes):
    val=wavefile.readframes(1)
    left=val[0:2]
#right=val[2:4]
    v=struct.unpack('h',left)[0]
    y[i]=v
Fs=framerate
specgram(y,NFFT=1024,Fs=Fs,noverlap=900)
show()
